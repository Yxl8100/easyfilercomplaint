```typescript
import { put } from '@vercel/blob'
import { prisma } from '@/lib/prisma'

export async function storeComplaintPdf(
  filingId: string,
  filingReceiptId: string,
  pdfBytes: Uint8Array
): Promise<string | null> {
  if (!process.env.BLOB_READ_WRITE_TOKEN) {
    console.warn('[pdf] BLOB_READ_WRITE_TOKEN not set -- skipping blob storage')
    return null
  }

  const blobPath = `complaints/${filingId}/EFC_${filingReceiptId}.pdf`

  const blob = await put(blobPath, Buffer.from(pdfBytes), {
    access: 'private',
    contentType: 'application/pdf',
    addRandomSuffix: false,
    allowOverwrite: true,
  })

  await prisma.filing.update({
    where: { id: filingId },
    data: { complaintPdfUrl: blob.url },
  })

  return blob.url
}
```

Key details:
- `access: 'private'` -- PDFs contain PII, must not be publicly accessible (per RESEARCH.md Pitfall 5)
- `addRandomSuffix: false` -- deterministic path for idempotent re-uploads
- `allowOverwrite: true` -- safe to re-run pipeline without duplicate files
- `Buffer.from(pdfBytes)` -- convert Uint8Array to Buffer for Vercel Blob put() body parameter
- The function takes filingId and filingReceiptId as separate string params (not the full Filing object) to keep the interface minimal
- Returns the blob URL string on success, or null if BLOB_READ_WRITE_TOKEN is absent
- Does NOT update Filing.status -- that is PIPE-03's responsibility in Phase 4
