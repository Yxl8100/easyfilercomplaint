```typescript
import { prisma } from '@/lib/prisma'
import { NextResponse } from 'next/server'
import { generateCPPAComplaintPdf } from '@/lib/cppa-pdf-generator'
import { put } from '@vercel/blob'

/**
 * GET /api/filings/[id]/cppa-pdf
 *
 * Returns the CPPA paper complaint PDF for the filing as a file download.
 *
 * Access model: UUID-only (per D-04/D-05 from Phase 10 CONTEXT.md).
 * The filing UUID (36-char random) IS the access token. No session check —
 * this allows guest filers to download their own form via a plain <a href> link
 * from the success page without requiring login.
 *
 * Storage: PDF is uploaded to Vercel Blob at a dedicated CPPA path
 * (complaints/cppa/{id}/CPPA_{receiptId}.pdf) so it does NOT overwrite the
 * CA AG PDF stored at complaints/{id}/EFC_{receiptId}.pdf. Filing.complaintPdfUrl
 * is NOT updated by this route — that field is reserved for the CA AG PDF.
 *
 * The route returns the PDF bytes regardless of whether Blob storage succeeds,
 * so a missing BLOB_READ_WRITE_TOKEN does not break the download.
 */
export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  // 1. Lookup Filing by UUID. UUID-only access — no session check (D-04/D-05).
  const filing = await prisma.filing.findUnique({
    where: { id: params.id },
  })

  if (!filing) {
    return NextResponse.json({ error: 'not_found' }, { status: 404 })
  }

  // 2. Generate PDF on demand (Plan 02 generator)
  const pdfBytes = await generateCPPAComplaintPdf(filing)

  // 3. Store in Vercel Blob at the dedicated CPPA path (Pitfall 3 — avoid AG PDF collision)
  //    Skip silently if BLOB_READ_WRITE_TOKEN is absent (matches store-complaint-pdf.ts pattern).
  if (process.env.BLOB_READ_WRITE_TOKEN) {
    const receiptId = filing.filingReceiptId ?? filing.id
    try {
      await put(
        `complaints/cppa/${filing.id}/CPPA_${receiptId}.pdf`,
        Buffer.from(pdfBytes),
        {
          access: 'private',
          contentType: 'application/pdf',
          addRandomSuffix: false,
          allowOverwrite: true,
        }
      )
    } catch (err) {
      // Blob storage failure must NOT block the consumer download.
      // Log and continue — PDF bytes still stream back to the client.
      console.error('[cppa-pdf] Blob storage failed:', err)
    }
  }

  // 4. Return PDF as file download (attachment, not inline — consumer must print and mail)
  const downloadId = filing.filingReceiptId ?? filing.id
  return new NextResponse(pdfBytes, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="CPPA_Complaint_${downloadId}.pdf"`,
    },
  })
}
```

**Critical reminders while implementing:**
- Do NOT add `import { auth } from '@/lib/auth'` or `const session = await auth()`. UUID is the access token. The test in `route.test.ts` reads the source file and asserts no `@/lib/auth` import exists.
- Do NOT call `storeComplaintPdf()` (the existing util) — it writes to a different blob path and updates `Filing.complaintPdfUrl`, which would break the CA AG PDF link. Use `put()` directly.
- Do NOT update `Filing.complaintPdfUrl` from this route. The CPPA PDF is regenerated fresh on each request — there is no need to persist its URL on the Filing record.
- The blob path MUST be `complaints/cppa/${filing.id}/CPPA_${receiptId}.pdf` exactly. Tests grep for `complaints/cppa/` and `CPPA_EFC-20260315-CPPDF.pdf`.
- Use `Content-Disposition: attachment` (NOT `inline`) — consumer needs to download and print, not view in-browser.
- Wrap `put()` in try/catch so a Blob failure (e.g., Vercel quota, network) does not 500 the route. The PDF must still stream to the client.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → API route | Untrusted `params.id` (path segment) — UUID lookup is the access check |
| API route → Prisma | `findUnique({ where: { id }})` — Prisma parameterizes; no SQL injection |
| API route → Vercel Blob | Server-side `put()` with `access: 'private'` — Blob URL not exposed to client |
| API route → Browser | PDF bytes streamed in response body with `attachment` disposition |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Elevation of Privilege | IDOR — guessing another filing's UUID to download their PDF | mitigate | UUIDs are 36-char random (cuid/v4); not enumerable in practice. Prisma `findUnique` returns null for any non-match → 404 response. Per D-04/D-05, UUID obscurity is the documented access model for both `/cppa-guide` (Phase 10) and this route. |
| T-11-02 | Information Disclosure | PDF returned to wrong user | mitigate | Same UUID = same filing = same intended recipient. PDF stored on Blob with `access: 'private'` — Blob URL is not exposed in the response (only PDF bytes are streamed). |
| T-11-03 | Information Disclosure | Filing.complaintPdfUrl (CA AG PDF link) being overwritten with CPPA PDF URL | mitigate | Acceptance criterion: route source does NOT contain `complaintPdfUrl:` as a Prisma update field. Blob path uses `complaints/cppa/` subdirectory, distinct from the AG PDF path. |
| T-11-04 | Denial of Service | Vercel Blob outage causing route to 500 | mitigate | `put()` wrapped in try/catch; Blob failure logs an error but the PDF bytes are still streamed. Route remains functional even if Blob is unavailable. |
| T-11-05 | Tampering | `params.id` containing path traversal characters used in Blob path | mitigate | Prisma's `findUnique({ where: { id }})` returns null for any value that isn't a real Filing UUID. By the time `params.id` reaches the Blob `put()` call, it has been validated against the database — no synthetic IDs reach the storage layer. |
| T-11-06 | Information Disclosure | BLOB_READ_WRITE_TOKEN absent from environment causing 500 errors | mitigate | Token presence guard `if (process.env.BLOB_READ_WRITE_TOKEN)` skips storage entirely when absent; route returns PDF bytes regardless. Verified by Wave 0 test "returns PDF bytes even when BLOB_READ_WRITE_TOKEN absent". |
| T-11-07 | Repudiation | No audit trail of who downloaded what | accept | Single-user filing model — only the UUID owner can access their own filing. Blob storage retains a copy with timestamp; no additional audit logging required for v2.0. |
