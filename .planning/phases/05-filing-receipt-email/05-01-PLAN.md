## Tasks

### Task 1: Create `src/lib/email-receipt.ts` — email builder and sender

**Files created:** `src/lib/email-receipt.ts`

**Action:**

Create a new file `src/lib/email-receipt.ts` with two exported functions:

**1. `buildReceiptEmailHtml(params)` — pure function, no side effects**

```typescript
export interface ReceiptEmailParams {
  filingReceiptId: string
  targetName: string
  agencyName: string
  paidAt: Date
  paymentAmount: number
  faxFailed: boolean
}

export function buildReceiptEmailHtml(params: ReceiptEmailParams): string
```

Returns an HTML string containing a simple, professional confirmation email. The HTML must include:

- Filing receipt ID (e.g., `EFC-20260401-ABC12`)
- Business name (`targetName`)
- Agency name (e.g., `California Attorney General`)
- Date filed — format `paidAt` as `Month D, YYYY` using `toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })`
- Amount paid — format `paymentAmount` as `$X.XX`
- If `faxFailed` is true, include a note: "Note: Fax delivery encountered an issue. Our team has been notified and will follow up."
- Footer: "This email is your filing receipt. Keep it for your records." and "Questions? Visit easyfilercomplaint.com"
- Branding: "EasyFilerComplaint" in the header

Use inline CSS for email compatibility (no external stylesheets). Keep it minimal — a white background, a centered container (max-width 600px), dark text. No images, no complex layout.

CRITICAL (EMAIL-05): The HTML must contain ZERO references to: DPW, PV Law, APFC, lawsuits, attorney. Do not include any of these words anywhere in the template. Do not reference law firms, attorneys, or litigation.

**2. `sendFilingReceiptEmail(filing, pdfBytes, faxFailed)` — sends email via Resend, updates DB**

```typescript
import { Resend } from 'resend'
import { prisma } from '@/lib/prisma'
import { getAgencyName } from '@/lib/agency-directory'
import type { Filing } from '@prisma/client'

const resend = new Resend(process.env.RESEND_API_KEY)

export async function sendFilingReceiptEmail(
  filing: Filing,
  pdfBytes: Uint8Array,
  faxFailed: boolean
): Promise<void>
```

Implementation details:

1. **Guard clauses** — throw if `!filing.filingReceiptId` or if consumer email is missing:
   ```typescript
   const filerInfo = filing.filerInfo as Record<string, string> | null
   const consumerEmail = filerInfo?.email
   if (!filing.filingReceiptId) throw new Error(`Filing ${filing.id} has no filingReceiptId`)
   if (!consumerEmail) throw new Error(`Filing ${filing.id} has no consumer email in filerInfo`)
   ```

2. **Build HTML** using `buildReceiptEmailHtml()`:
   ```typescript
   const html = buildReceiptEmailHtml({
     filingReceiptId: filing.filingReceiptId,
     targetName: filing.targetName,
     agencyName: getAgencyName('ca_ag'),
     paidAt: filing.paidAt ?? new Date(),
     paymentAmount: Number(filing.paymentAmount ?? 0),
     faxFailed,
   })
   ```

3. **Send via Resend** (EMAIL-01, EMAIL-02, EMAIL-04):
   ```typescript
   await resend.emails.send({
     from: 'EasyFilerComplaint <noreply@easyfilercomplaint.com>',
     to: [consumerEmail],
     subject: `Your EasyFilerComplaint Filing Receipt — ${filing.filingReceiptId}`,
     html,
     attachments: [
       {
         filename: `EFC_Filing_${filing.filingReceiptId}.pdf`,
         content: Buffer.from(pdfBytes),
         contentType: 'application/pdf',
       },
     ],
   })
   ```

4. **Update DB** (EMAIL-06):
   ```typescript
   await prisma.filing.update({
     where: { id: filing.id },
     data: { receiptEmailSentAt: new Date() },
   })
   ```

5. **Log success:**
   ```typescript
   console.log(`[email] Receipt email sent to ${consumerEmail} for filing ${filing.filingReceiptId}`)
   ```

```typescript
import { describe, it, expect } from 'vitest'
import { buildReceiptEmailHtml } from '@/lib/email-receipt'

describe('buildReceiptEmailHtml', () => {
  const params = {
    filingReceiptId: 'EFC-20260401-ABC12',
    targetName: 'Acme Corp',
    agencyName: 'California Attorney General',
    paidAt: new Date('2026-04-01T12:00:00Z'),
    paymentAmount: 1.99,
    faxFailed: false,
  }

  const html = buildReceiptEmailHtml(params)

  // EMAIL-05: Prohibited strings
  const PROHIBITED = ['DPW', 'PV Law', 'APFC', 'lawsuits', 'attorney']
  PROHIBITED.forEach((word) => {
    it(`does not contain prohibited string "${word}" (case-insensitive)`, () => {
      expect(html.toLowerCase()).not.toContain(word.toLowerCase())
    })
  })

  // EMAIL-03: Required content
  it('contains filing receipt ID', () => {
    expect(html).toContain('EFC-20260401-ABC12')
  })

  it('contains business name', () => {
    expect(html).toContain('Acme Corp')
  })

  it('contains agency name', () => {
    expect(html).toContain('California Attorney General')
  })

  it('contains formatted payment amount', () => {
    expect(html).toContain('$1.99')
  })

  it('contains formatted date', () => {
    // toLocaleDateString('en-US', ...) produces "April 1, 2026"
    expect(html).toContain('2026')
  })

  it('does not contain fax failure note when faxFailed=false', () => {
    expect(html).not.toContain('Fax delivery encountered an issue')
  })

  it('contains fax failure note when faxFailed=true', () => {
    const failedHtml = buildReceiptEmailHtml({ ...params, faxFailed: true })
    expect(failedHtml).toContain('Fax delivery encountered an issue')
  })
})
```
