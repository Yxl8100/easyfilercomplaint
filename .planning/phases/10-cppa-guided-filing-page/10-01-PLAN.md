Create `src/components/CopyButton.tsx`:

```tsx
'use client'

import { useState } from 'react'

interface CopyButtonProps {
  text: string
}

export function CopyButton({ text }: CopyButtonProps) {
  const [copied, setCopied] = useState(false)

  async function handleCopy() {
    await navigator.clipboard.writeText(text)
    setCopied(true)
    setTimeout(() => setCopied(false), 1500)
  }

  return (
    <button
      onClick={handleCopy}
      className="font-mono text-[9px] uppercase tracking-[0.1em] border border-border rounded-[6px] px-3 py-1 text-text-mid hover:text-text transition-colors"
      aria-label={copied ? 'Copied to clipboard' : 'Copy to clipboard'}
    >
      {copied ? 'Copied!' : 'Copy'}
    </button>
  )
}
```

```tsx
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { render, screen, fireEvent } from '@testing-library/react'
import { CopyButton } from './CopyButton'

describe('CopyButton', () => {
  beforeEach(() => {
    // Mock clipboard API — not available in jsdom
    Object.defineProperty(navigator, 'clipboard', {
      value: { writeText: vi.fn().mockResolvedValue(undefined) },
      writable: true,
      configurable: true,
    })
  })

  afterEach(() => {
    vi.restoreAllMocks()
  })

  it('CPGDE-03: renders a button with "Copy" label', () => {
    render(<CopyButton text="some text" />)
    expect(screen.getByRole('button', { name: /copy/i })).toBeTruthy()
  })

  it('CPGDE-03: calls navigator.clipboard.writeText with the exact text prop on click', async () => {
    render(<CopyButton text="my complaint text" />)
    const btn = screen.getByRole('button', { name: /copy/i })
    fireEvent.click(btn)
    expect(navigator.clipboard.writeText).toHaveBeenCalledWith('my complaint text')
  })

  it('shows "Copied!" label after click', async () => {
    vi.useFakeTimers()
    render(<CopyButton text="hello" />)
    fireEvent.click(screen.getByRole('button', { name: /copy/i }))
    // Allow microtask queue to flush
    await Promise.resolve()
    expect(screen.getByRole('button').textContent).toContain('Copied!')
    vi.useRealTimers()
  })
})
```
